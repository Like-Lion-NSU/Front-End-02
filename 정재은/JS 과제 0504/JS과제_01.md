1. 키
```js
        <script>
        const height = prompt("키 입력:");

        const array = height.split(" "); //공백 으로 잘라서 배열로 저장한다.

        const num = array.map(Number); //각각 숫자로 바꾸기

        let result = true;
      
        for(let i=0; i<=num.length-1; i++){ 
            if(num[i]>num[i+1]){
                result= false;
                break;
            }            
        }
        if(result)
            alert("YES")
        else
            alert("NO")

          /* 처음 도전했던 식 -> 반복문 안에 alert가 있으니 처음 두개의 yes/no만 판명하는 듯
        for(let i=0; i<=num.length-1; i++){ 
            if(num[i]>num[i+1]){
                alert("NO");
                break;
            }            
            else
                alert("YES");
                break;
        } 
        */ 
    </script>
    
```
2. 구구단
```js
    <script>
        const dan=Number(prompt("단을 입력하세요"));
        let result=""; //처음엔 공백
        for(let i=1; i<=9; i++){
            result=result+i*dan+" " //result에 기존 result값과 i*dan, 띄어쓰기 대입
        }
        console.log(result);
    </script>
```
